            function disp(str)
            {
                var r = str.value;
                var len = r.length
                document.write("length of the string =" + len + "<br>");
                var ch = r.charAt(3);
                document.write("char=" + ch +"<br>");
                var ing = r.indexOf("e");
                document.write("position of e =" + ing +"<br>");
                var sub = r.substring(2,4);
                document.write("substring ="+ sub + "<br>");
                var low = r.toLowerCase();
                document.write("LowerCase ="+ low + "<br>");
                var high = r.toUpperCase();
                document.write("UpperCase ="+ high + "<br>");
            }
        </script>
