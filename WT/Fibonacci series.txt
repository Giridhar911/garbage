Fibonacci series
<html>
    <head>
        <title>document</title>
        <script>
            var n = parseInt(prompt('ENTER THE NUMBER OF TERMS:'));
            var n1 = 0, n2 = 1, nextTerm;
            document.write('fibonacci series:<br/>');
            for(var i=1;i<=n;i++)
            {
                document.write(n1+'<br/>');
                nextTerm=n1+n2;
                n1=n2;
                n2=nextTerm;
            }
        </sfibcript>
    </head>
</html>
